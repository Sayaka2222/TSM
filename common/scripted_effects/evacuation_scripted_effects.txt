ger_fall_of_france_payout = {

  if = { limit = { has_global_flag = fall_of_france_payout_done } set_global_flag = fall_of_france_payout_done }

  every_state = {
    limit = { has_state_flag = mainland_france_state }
    set_building_level = { type = arms_factory level = 0 }
    set_building_level = { type = industrial_complex level = 0 }
  }

  GER = {
    for_loop_effect = {
      start = 0
      end = 30
      random_owned_state = {
        limit = { has_state_flag = mainland_france_state }
        add_extra_state_shared_building_slots = 1
        add_building_construction = { type = arms_factory level = 1 instant_build = yes }
      }
    }
    for_loop_effect = {
      start = 0
      end = 30
      random_owned_state = {
        limit = { has_state_flag = mainland_france_state }
        add_extra_state_shared_building_slots = 1
        add_building_construction = { type = industrial_complex level = 1 instant_build = yes }
      }
    }
  }

  FRA = {
    for_loop_effect = {
      start = 0
      end = 40
      random_owned_state = {
        limit = { has_state_flag = french_africa_state }
        add_extra_state_shared_building_slots = 1
        add_building_construction = { type = arms_factory level = 1 instant_build = yes }
      }
    }
    for_loop_effect = {
      start = 0
      end = 20
      random_owned_state = {
        limit = { has_state_flag = french_africa_state }
        add_extra_state_shared_building_slots = 1
        add_building_construction = { type = industrial_complex level = 1 instant_build = yes }
      }
    }
  }

  GER = {
    set_temp_variable = { m = fra_hold_months_total }
    if = { limit = { check_variable = { m > 8 } } set_temp_variable = { m = 8 } }
    if = { limit = { check_variable = { m < 0 } } set_temp_variable = { m = 0 } }

    set_temp_variable = { extra_mils = m }
    multiply_temp_variable = { var = extra_mils value = 5 }

    set_temp_variable = { extra_civs = 0 }

    set_temp_variable = { a = m }
    if = { limit = { check_variable = { a > 4 } } set_temp_variable = { a = 4 } }
    multiply_temp_variable = { var = a value = 3 }
    add_to_temp_variable = { var = extra_civs value = a }

    set_temp_variable = { b = m }
    add_to_temp_variable = { var = b value = -4 }
    if = { limit = { check_variable = { b < 0 } } set_temp_variable = { b = 0 } }
    multiply_temp_variable = { var = b value = 2 }
    add_to_temp_variable = { var = extra_civs value = b }
  }

  GER = {
    for_loop_effect = {
      start = 0
      end = var:extra_mils
      random_state = {
        limit = { has_state_flag = french_africa_state }
        add_extra_state_shared_building_slots = 1
        add_building_construction = { type = arms_factory level = 1 instant_build = yes }
      }
    }
    for_loop_effect = {
      start = 0
      end = var:extra_civs
      random_state = {
        limit = { has_state_flag = french_africa_state }
        add_extra_state_shared_building_slots = 1
        add_building_construction = { type = industrial_complex level = 1 instant_build = yes }
      }
    }
  }
}
